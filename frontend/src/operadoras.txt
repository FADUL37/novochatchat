// Theme Toggle
function toggleTheme() {
  isDarkTheme = !isDarkTheme;
  const body = document.body;
  const icon = document.getElementById('theme-icon');
  
  if (isDarkTheme) {
    body.classList.remove('light-theme');
    icon.className = 'fas fa-moon';
  } else {
    body.classList.add('light-theme');
    icon.className = 'fas fa-sun';
  }
}

// Audio Recording
function toggleRecording() {
  const recordBtn = document.getElementById('recordBtn');
  const visualizer = document.getElementById('audioVisualizer');
  
  if (!isRecording) {
    startRecording();
    recordBtn.innerHTML = '<i class="fas fa-stop"></i>';
    recordBtn.classList.add('recording');
    visualizer.style.display = 'flex';
  } else {
    stopRecording();
    recordBtn.innerHTML = '<i class="fas fa-microphone"></i>';
    recordBtn.classList.remove('recording');
    visualizer.style.display = 'none';
  }
}

async function startRecording() {
  try {
    const stream = await navigator.mediaDevices.getUserMedia({ audio: true });
    mediaRecorder = new MediaRecorder(stream);
    audioChunks = [];
    
    mediaRecorder.ondataavailable = (event) => {
      audioChunks.push(event.data);
    };
    
    mediaRecorder.onstop = () => {
      const audioBlob = new Blob(audioChunks, { type: 'audio/wav' });
      const audioUrl = URL.createObjectURL(audioBlob);
      sendAudioMessage(audioUrl);
    };
    
    mediaRecorder.start();
    isRecording = true;
    showNotification('ğŸ¤ GravaÃ§Ã£o iniciada!');
  } catch (error) {
    console.error('Erro ao acessar microfone:', error);
    showNotification('âŒ Erro ao acessar microfone');
  }
}

function stopRecording() {
  if (mediaRecorder && isRecording) {
    mediaRecorder.stop();
    mediaRecorder.stream.getTracks().forEach(track => track.stop());
    isRecording = false;
    showNotification('ğŸ¤ GravaÃ§Ã£o finalizada!');
  }
}

function handleGifSelect(event) {
  const file = event.target.files[0];
  if (file && file.type === 'image/gif') {
    const reader = new FileReader();
    reader.onload = (e) => {
      sendGifMessage(e.target.result);
    };
    reader.readAsDataURL(file);
  }
}

function sendChatbotMessage() {
  const input = document.getElementById('chatbot-input');
  const message = input.value.trim().toLowerCase();
  
  if (message) {
    addChatbotMessage(`ğŸ‘¤ VocÃª: ${input.value}`);
    input.value = '';
    
    showTypingIndicator();
    
    setTimeout(() => {
      hideTypingIndicator();
      generateChatbotResponse(message);
    }, 1000);
  }
}

function addChatbotMessage(message) {
  const messagesContainer = document.getElementById('chatbot-messages');
  const messageDiv = document.createElement('div');
  messageDiv.className = 'chatbot-message';
  messageDiv.innerHTML = message.replace(/\n/g, '<br>');
  messagesContainer.appendChild(messageDiv);
  messagesContainer.scrollTop = messagesContainer.scrollHeight;
}

function showTypingIndicator() {
  const messagesContainer = document.getElementById('chatbot-messages');
  const typingDiv = document.createElement('div');
  typingDiv.className = 'chatbot-message typing-indicator';
  typingDiv.id = 'typing-indicator';
  typingDiv.innerHTML = 'ğŸ¤– Digitando<span class="dots">...</span>';
  messagesContainer.appendChild(typingDiv);
  messagesContainer.scrollTop = messagesContainer.scrollHeight;
}

function hideTypingIndicator() {
  const typingIndicator = document.getElementById('typing-indicator');
  if (typingIndicator) {
    typingIndicator.remove();
  }
}

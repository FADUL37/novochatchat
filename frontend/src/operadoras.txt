// Funcionalidade do chat original
function setNickname() {
  const input = document.getElementById('nicknameInput');
  username = input.value.trim();
  if (username) {
    socket.emit('setUsername', username);

function sendMessage() {
  const input = document.getElementById('m');
  const message = input.value.trim();
  if (message) {
    socket.emit('chat message', message);
    input.value = '';
    addMessageToChat(`${username}: ${message}`);
  }
}

function addMessageToChat(message) {
  const item = document.createElement('li');
  item.className = 'message-item';

// Theme Toggle
function toggleTheme() {
  isDarkTheme = !isDarkTheme;
  const body = document.body;
  const icon = document.getElementById('theme-icon');
  
  if (isDarkTheme) {
    body.classList.remove('light-theme');
    icon.className = 'fas fa-moon';
  } else {
    body.classList.add('light-theme');
    icon.className = 'fas fa-sun';
  }
}

// Audio Recording
function toggleRecording() {
  const recordBtn = document.getElementById('recordBtn');
  const visualizer = document.getElementById('audioVisualizer');
  
  if (!isRecording) {
    startRecording();
    recordBtn.innerHTML = '<i class="fas fa-stop"></i>';
    recordBtn.classList.add('recording');
    visualizer.style.display = 'flex';
  } else {
    stopRecording();
    recordBtn.innerHTML = '<i class="fas fa-microphone"></i>';
    recordBtn.classList.remove('recording');
    visualizer.style.display = 'none';
  }
}

async function startRecording() {
  try {
    const stream = await navigator.mediaDevices.getUserMedia({ audio: true });
    mediaRecorder = new MediaRecorder(stream);
    audioChunks = [];
    
    mediaRecorder.ondataavailable = (event) => {
      audioChunks.push(event.data);
    };
    
    mediaRecorder.onstop = () => {
      const audioBlob = new Blob(audioChunks, { type: 'audio/wav' });
      const audioUrl = URL.createObjectURL(audioBlob);
      sendAudioMessage(audioUrl);
    };
    
    mediaRecorder.start();
    isRecording = true;
    showNotification('🎤 Gravação iniciada!');
  } catch (error) {
    console.error('Erro ao acessar microfone:', error);
    showNotification('❌ Erro ao acessar microfone');
  }
}

function stopRecording() {
  if (mediaRecorder && isRecording) {
    mediaRecorder.stop();
    mediaRecorder.stream.getTracks().forEach(track => track.stop());
    isRecording = false;
    showNotification('🎤 Gravação finalizada!');
  }
}

function sendAudioMessage(audioUrl) {
  const audioElement = document.createElement('audio');
  audioElement.controls = true;
  audioElement.src = audioUrl;
  audioElement.style.maxWidth = '200px';
  audioElement.style.borderRadius = '12px';
  
  const message = `${username} enviou um áudio:`;
  const item = document.createElement('li');
  item.className = 'message-item';
  item.innerHTML = `
    <div>
      <div style="margin-bottom: 8px;">
        <span>🎵 ${message}</span>
        <small style="margin-left: 10px; opacity: 0.7;">${new Date().toLocaleTimeString()}</small>
      </div>
    </div>
  `;
  item.appendChild(audioElement);
  
  document.getElementById('messages').appendChild(item);
  item.scrollIntoView({ behavior: 'smooth' });
  
  // Emit audio message to other users
  socket.emit('audio message', { username, audioUrl, timestamp: new Date().toLocaleTimeString() });
}

// GIF Selection
function selectGif() {
  document.getElementById('gifInput').click();
}

function handleGifSelect(event) {
  const file = event.target.files[0];
  if (file && file.type === 'image/gif') {
    const reader = new FileReader();
    reader.onload = (e) => {
      sendGifMessage(e.target.result);
    };
    reader.readAsDataURL(file);
  }
}

// Notification System
function showNotification(message) {
  const notification = document.createElement('div');
  notification.className = 'notification';
  notification.innerHTML = `
    <div style="display: flex; align-items: center; gap: 10px;">
      <i class="fas fa-bell"></i>
      <span>${message}</span>
    </div>
  `;
  
  document.body.appendChild(notification);
  
  setTimeout(() => {
    notification.style.animation = 'notificationSlide 0.3s ease reverse';
    setTimeout(() => {
      document.body.removeChild(notification);
    }, 300);
  }, 3000);
}

// Chatbot functionality
function toggleChatbot() {
  const chatbot = document.getElementById('chatbot');
  const chatbotMessages = document.getElementById('chatbot-messages');
  
  chatbotOpen = !chatbotOpen;
  
  if (chatbotOpen) {
    chatbot.style.display = 'flex';
    if (chatbotMessages.children.length === 0) {
      addChatbotMessage('🤖 Olá! Sou seu assistente virtual. Como posso ajudá-lo hoje?\n\n📋 Digite "operadoras" para ver a lista completa\n🔍 Digite o nome de uma operadora para informações específicas\n📞 Digite "contato [operadora]" para informações de contato');
    }
  } else {
    chatbot.style.display = 'none';
  }
}

function sendChatbotMessage() {
  const input = document.getElementById('chatbot-input');
  const message = input.value.trim().toLowerCase();
  
  if (message) {
    addChatbotMessage(`👤 Você: ${input.value}`);
    input.value = '';
    
    showTypingIndicator();
    
    setTimeout(() => {
      hideTypingIndicator();
      generateChatbotResponse(message);
    }, 1000);
  }
}

function addChatbotMessage(message) {
  const messagesContainer = document.getElementById('chatbot-messages');
  const messageDiv = document.createElement('div');
  messageDiv.className = 'chatbot-message';
  messageDiv.innerHTML = message.replace(/\n/g, '<br>');
  messagesContainer.appendChild(messageDiv);
  messagesContainer.scrollTop = messagesContainer.scrollHeight;
}

function showTypingIndicator() {
  const messagesContainer = document.getElementById('chatbot-messages');
  const typingDiv = document.createElement('div');
  typingDiv.className = 'chatbot-message typing-indicator';
  typingDiv.id = 'typing-indicator';
  typingDiv.innerHTML = '🤖 Digitando<span class="dots">...</span>';
  messagesContainer.appendChild(typingDiv);
  messagesContainer.scrollTop = messagesContainer.scrollHeight;
}

function hideTypingIndicator() {
  const typingIndicator = document.getElementById('typing-indicator');
  if (typingIndicator) {
    typingIndicator.remove();
  }
}

function generateChatbotResponse(message) {
  let response = '';
  
  if (message.includes('operadoras') || message.includes('lista')) {
    response = '📋 **Operadoras Disponíveis:**\n\n';
    Object.keys(operadorasData).forEach(operadora => {
      response += `🔹 ${operadora}\n`;
    });
    response += '\n💡 Digite o nome de qualquer operadora para mais informações!';
  }
  else if (message.includes('contato')) {
    const operadoraMatch = message.match(/contato\s+(\w+)/i);
    if (operadoraMatch) {
      const operadoraNome = operadoraMatch[1].toUpperCase();
      const operadora = Object.keys(operadorasData).find(op => 
        op.toUpperCase().includes(operadoraNome) || operadoraNome.includes(op.toUpperCase())
      );
      
      if (operadora) {
        const data = operadorasData[operadora];
        response = `📞 **Contatos - ${operadora}:**\n\n`;
        if (data.telefone) response += `${data.telefone}\n`;
        if (data.whatsapp) response += `${data.whatsapp}\n`;
        if (data.email) response += `${data.email}\n`;
        if (data.site) response += `${data.site}\n`;
      } else {
        response = '❌ Operadora não encontrada. Digite "operadoras" para ver a lista completa.';
      }
    }
  }
  else {
    // Buscar por operadora específica
    const operadora = Object.keys(operadorasData).find(op => 
      message.includes(op.toLowerCase()) || op.toLowerCase().includes(message)
    );
    
    if (operadora) {
      const data = operadorasData[operadora];
      response = `📊 **Informações - ${operadora}:**\n\n`;
      
      if (data.tecnologias) response += `**Tecnologias:** ${data.tecnologias}\n\n`;
      if (data.planos || data.planos_residenciais) {
        response += `**Planos:** ${data.planos || data.planos_residenciais}\n\n`;
      }
      if (data.cidades) response += `**Cidades:** ${data.cidades}\n\n`;
      if (data.telefone) response += `**Contato:** ${data.telefone}\n\n`;
      
      response += '💡 Digite "contato ' + operadora.toLowerCase() + '" para informações completas de contato!';
    } else {
      response = '🤔 Não entendi sua pergunta. Tente:\n\n📋 "operadoras" - para ver todas as operadoras\n🔍 Digite o nome de uma operadora\n📞 "contato [operadora]" - para contatos';
    }
  }
  
  addChatbotMessage('🤖 ' + response);
}
